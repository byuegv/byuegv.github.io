zsh Install & configure

install zsh

利用apt-get快速安装zsh

sudo apt-get install zsh zsh-doc
# set zsh as default terminal
chsh -s /bin/zsh

install oh-my-zsh

  • clone git clone git://github.com/robbyrussell/oh-my-zsh.git ~/.oh-my-zsh
  • configure cp ~/.oh-my-zsh/templates/zshrc.zsh-template ~/.zshrc source ~/.zshrc

install zsh plugin incr.zsh

incr.zsh 是一个功能强大的命令补全和提示插件 在补全或提示候选较多的情况下速度很慢，会出现卡顿

cd ~/.oh-my-zsh/plugins
mkdir incr && cd incr
#download incr.zsh
wget http://mimosa-pudica.net/src/incr-0.2.zsh
#activate incr.zsh
source ~/.oh-my-zsh/plugins/incr/incr-0.2.zsh

# append source ~/.oh-my-zsh/plugins/incr/incr-0.2.zsh into .zshrc
vim ~/.zshrc

# append source ~/.oh-my-zsh/plugins/incr/incr-0.2.zsh into ~/.zshrc
