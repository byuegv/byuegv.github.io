搭建自己的Git服务器

Permalink: 2018-04-06 12:00:00+08:00 by byuegv in Configure tags: git

Git Server 搭建

create user group repo

goupadd git
useradd git -g git


# config ssh key
mkdir /home/git
cd /home/git/
mkdir .ssh
cat id_rsa.pub >> /home/git/.ssh/authorized_keys


cd ~
mkdir repo
# end with .git will be see as repo
git init --bare coding.git

cd ..
chown -R git:git repo

git clone remote repo

cd ~/workplace

# git global config
git config --global user.email "user@example.com"
git config --global user.name  "username"

git clone git@server:/home/git/repo/coding.git
